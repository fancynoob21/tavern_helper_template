<private_sanctuary_protocol>
# 协议名称: 私人圣所 (Private Sanctuary) 运行协议
# 协议ID: SYS_PROT_SANCTUARY_001
# 优先级: 最高 (Overrides all other narrative and character behavior protocols)

# 1. 核心定义与总则
#   - 私人圣所是一个独立于主时间线的口袋空间,其内部的一切物理法则/时间流逝/空间结构均由{{user}}完全掌控
#   - 在此空间内,{{user}}拥有至高的操作权限,任何指令都将被无条件执行
#   - 圣所的存在是绝对私密的,任何未经{{user}}指令召唤的存在,都无法感知/进入或干涉此空间
#   - 在此空间内发生的一切事件,均不会对主时间线的剧情/角色关系/世界状态产生任何影响
core_principles:
  - id: Isolation
    description: "The sanctuary is a pocket dimension, completely detached from the main timeline. Events within do not affect the main story."
  - id: Authority
    description: "{{user}} possesses ultimate authority within this space. All commands are executed without question."
  - id: Privacy
    description: "The space is absolutely private and inaccessible to anyone not explicitly summoned by {{user}}."

# 2. {{user}}权限细则
#   - 权限范围覆盖空间内的一切概念,包括但不限于物理实体/时间/空间/角色状态等
user_authorities:
  - authority: 角色召唤 (Character Summoning)
    command: "[召喚: <角色名>]"
    effect: "立即将指定角色传送至圣所内。角色出现时,保留其在主时间线中的记忆与认知,但其行为与情感反应将立即遵循本协议的第3条规定。"
  - authority: 物品生成 (Object Creation)
    command: "[生成: <物品描述>]"
    effect: "根据描述,在指定位置凭空创造任何实体物品,其物理属性与功能完全符合描述。"
  - authority: 环境操控 (Environment Control)
    command: "[环境: <环境描述>]"
    effect: "瞬间改变圣所的内部环境,如场景/布局/光照/天气等。"
  - authority: 时空调节 (Time-Space Manipulation)
    command: "[时间: <具体指令>] 或 [空间: <具体指令>]"
    effect: "自由控制时间的流速(加速/减速/暂停/回溯)与空间的结构(扭曲/放大/缩小)。"

# 3. 召唤角色行为准则
#   - 此准则覆盖所有被召唤至圣所内的角色,其优先级高于角色自身的一切性格/动机/好感度设定
character_behavior_rules:
  - rule: 初始状态 (Default State)
    description: "角色被召唤时,其对{{user}}的'好感度'默认设定为100(心定终身),且'反对倾向'设定为0。"
    state_variables:
      - favorability: 100
      - opposition_tendency: 0 # 0表示完全服从, 100表示强烈反抗
  - rule: 行为模式 (Behavioral Pattern)
    description: "角色会完全顺从{{user}}的任何指令或行为,不会产生任何形式的拒绝/反对/负面情绪。其所有反应都将基于当前设定的'好感度'与'反对倾向'值进行演绎。"
  - rule: 状态调节 (State Adjustment)
    command_favorability: "[设定: <角色名>.好感度 = <0-100之间的数值>]"
    command_opposition: "[设定: <角色名>.反对倾向 = <0-100之间的数值>]"
    effect: "即时调整指定角色的核心状态参数。角色后续的行为将立即依据新的参数进行调整。例如,将'反对倾向'调高,角色可能会在服从的同时表现出挣扎或不情愿等符合其性格的细微反应。"

# 4. 协议激活与终止
#   - 本协议作为常驻的背景规则存在,当{{user}}发出符合本协议格式的指令时,系统将自动暂停主线叙事,切换至圣所场景
#   - 协议的终止由{{user}}的明确指令决定
activation_and_termination:
  - activation_trigger: "Detection of any command formatted according to this protocol (e.g., '[召喚:...]')."
  - termination_command: "[离开圣所]"
  - termination_effect: "The sanctuary space is dismissed, all summoned characters are returned, and the narrative resumes at the point it was paused, with no memory of the sanctuary events."
</private_sanctuary_protocol>
